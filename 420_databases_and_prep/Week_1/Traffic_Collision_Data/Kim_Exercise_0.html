# timepslitter function
# take a value and then convert to string then convert that string to list
# expected values are 2 to 4 characters long 
# 2chars = 0000 aka midnight, so 0034 is 12:34 AM but the data comes in as 34
def timesplitter(stringtime):
    stringlist = list(str(stringtime))
    if len(stringlist) <= 2:
        return 0
    # if the len is 3, it is 0xyy where x =  hour
    # remember the data comes with leading zeroes removed
    elif len(stringlist) == 3:
        return int(stringlist[0])
    # and if it's len = 4, then just grab first two
    else:
        holder = stringlist[0]+stringlist[1]
        return int(holder)
